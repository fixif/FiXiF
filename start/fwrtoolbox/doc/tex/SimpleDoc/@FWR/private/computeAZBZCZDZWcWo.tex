\begin{command}[@FWR/computeAZBZCZDZWcWo]{computeAZBZCZDZWcWo}	\desc{Purpose}Compute the $A_Z$, $B_Z$, $C_Z$ and $D_Z$ matrices and the gramians of a FWR object	\desc{Syntax}\matlab{R=updateAZBZCZDZWoWc(R)}	\desc{Parameters}		\begin{tabular}{l@{\ :\ }p{9cm}}\matlab{R} &  FWR object\\		\end{tabular}	\desc{Description}\begin{center}\I{Internal function}\end{center}Compute the matrices $A_Z$, $B_Z$, $C_Z$, $D_Z$ and the controllability and observability gramians associated with a FWR object.\\$A_Z\in\Rbb{n}{n}$, $B_Z\in\Rbb{n}{m}$, $C_Z\in\Rbb{p}{n}$ and $D_Z\in\Rbb{p}{m}$ are given by:\begin{align}A_Z &= KJ^{-1}M+P,   &  B_Z &= KJ^{-1}N+Q, \\C_Z &= LJ^{-1}M+R, &  	D_Z &= LJ^{-1}N+S.\end{align}and the gramians are the solutions to the Lyapunov equations\begin{eqnarray}W_c &=& A_Z W_c A_Z^\top + B_Z B_Z^\top \\W_o &=& A_Z^\top W_o A_Z + C_Z^\top C_Z\end{eqnarray}\end{command}
