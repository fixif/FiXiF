\begin{command}[@FWR/mylyap]{mylyap}	\desc{Purpose}Solve the continuous-time Lyapunov equationsadapted from \matlab{lyap.m} (S.N. Bangert The MathWorks, Inc.)	\desc{Syntax}\matlab{X = mylyap(A, B, C, ua, ta)}	\desc{Parameters}		\begin{tabular}{l@{\ :\ }p{9cm}}\matlab{X} &  solution of the continuous-time Lyapunov equation\\\matlab{A,B,C} &  parameters of the equation                   \\\matlab{ua,ta} &  precomputed values                           \\		\end{tabular}	\desc{Description}\begin{center}\I{Internal function}\end{center}This function executes the same algorithm described in \matlab{lyap.m} (S.N. Bangert The MathWorks, Inc.), exceptthat the values \matlab{ua} and \matlab{ta} are already computed. It permits to save some computational time when a lot ofLyapunov equations have to be solved with the same value $A$.\desc{See also}\funcName[@FWR/wprodnormSISO]{w\_prod\_norm\_SISO}\end{command}
