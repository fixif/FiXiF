\begin{command}[rhoDFIIt2FWS]{rhoDFIIt2FWS}	\desc{Purpose}Build a $\rho$-DFIIt structuration($\rho$ Direct Form II transposed, according to Li and Zhao's work)this realization can be $L_2$-scaled or not	\desc{Syntax}\matlab{[S1, S2] = rhoDFIIt2FWS( H, gamma, isGammaExact, delta, isDeltaExact)}	\desc{Parameters}		\begin{tabular}{l@{\ :\ }p{9cm}}\matlab{R1} &  $\rho$DFIIt structuration (FWS object)                                                     \\\matlab{R2} &  equivalent $q$-state-space (sparse realization) FWS object                                 \\\matlab{H} &  \matlab{tf} object                                                                                 \\\matlab{gamma } &  vector of parameters $\gamma_k$                                                        \\\matlab{isGammaExact } &  (default: true) boolean to express if $\gamma$ are exactly represented or not   \\\matlab{delta } &  vector or parameters $\Delta_k$                                                        \\ & if they are not given, a $L_2$-scaling is performed ($\Delta_k$ then are induced)              \\\matlab{isDeltaExact } &  (default: false) boolean to express if $\Delta_k$ are exactly represented or not\\		\end{tabular}\desc{See also}\funcName[rhoDFIIt2FWR]{rhoDFIIt2FWR}	\desc{References}\cite{Li04b} G. Li and Z. Zhao. On the generalized DFIIt structureand its state-space realization in digital filter implementation. IEEE Trans. on Circuits and Systems, 51(4):769--778, April 2004\end{command}
