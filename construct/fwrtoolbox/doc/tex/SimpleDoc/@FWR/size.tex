\begin{command}[@FWR/size]{size}	\desc{Purpose}Return the size of the FW Realization	\desc{Syntax}\matlab{lmnp = size(R)     }\\\matlab{[l,m,n,p] = size(R)}	\desc{Parameters}		\begin{tabular}{l@{\ :\ }p{9cm}}\matlab{lmnp} &  vector [l,m,n,p]         \\\matlab{l} &  nb of intermediate variables\\\matlab{m} &  nb of inputs                \\\matlab{n} &  nb of states                \\\matlab{p} &  nb of outputs               \\\matlab{R} &  FWR object                  \\		\end{tabular}	\desc{Description}Overload of the classical \matlab{size} function.\end{command}
