\begin{command}[@FWR/tf]{tf}	\desc{Purpose}Convert a FWR object into a tf object (transfer function)	\desc{Syntax}\matlab{H = tf(R,Te)}	\desc{Parameters}		\begin{tabular}{l@{\ :\ }p{9cm}}\matlab{H} &  tf object          \\\matlab{R} &  FWR object         \\\matlab{Te} &  period (default=1)\\		\end{tabular}	\desc{Description}Give the transfer function of the realization. It is defined with matrices$A_Z$, $B_Z$, $C_Z$ and $D_Z$ (see eq. \eqref{eq:defAZandBZ} and \eqref{eq:defCZandDZ}) by:\begin{equation}H:z\mapsto C_Z \pa{zI_n-A_Z}^{-1}B_Z + D_Z\end{equation}\desc{See also}\funcName[@FWR/ss]{ss}, \funcName[@FWS/tf]{tf}\end{command}
