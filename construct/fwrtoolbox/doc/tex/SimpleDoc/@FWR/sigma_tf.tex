\begin{command}[@FWR/sigmatf]{sigma\_tf}	\desc{Purpose}Compute the open-loop transfer function error $\sigma_{\Delta H}^2$)	\desc{Syntax}\matlab{[e, eZ]  = sigma\_tf(R)}	\desc{Parameters}		\begin{tabular}{l@{\ :\ }p{9cm}}\matlab{e} &  transfer function error       \\\matlab{e} &  transfer function error matrix\\\matlab{R} &  FWR object                    \\		\end{tabular}	\desc{Description}The open-loop transfer function error is defined by\begin{equation}\sigma_{\Delta H}^2 \triangleq \frac{1}{2\pi} \int_0^{2\pi} E\left\{ \abs{\Delta H \pa{ e^{j\omega} }}^2 \right\} d\omega\end{equation}where $E\{.\}$ is the mean operator.\desc{See also}\funcName[@FWR/MsensH]{MsensH}\end{command}
