\begin{command}[@FWR/computeJtoS]{computeJtoS}	\desc{Purpose}Compute the $J$, $K$, $L$, $M$, $N$, $P$, $Q$, $R$, $S$, $W_J$, ..., $W_J$,  of a FWR objectfrom the $Z$ and $W_Z$ matrices of this object	\desc{Syntax}\matlab{R=computeJtoS(R)}	\desc{Parameters}		\begin{tabular}{l@{\ :\ }p{9cm}}\matlab{R} &  FWR object\\		\end{tabular}	\desc{Description}\begin{center}\I{Internal function}\end{center}This function updates the matrices $J$, $K$, $L$, $M$, $N$, $P$, $Q$, $R$, $Q$, $W_J$, $W_K$, $W_L$,$W_M$, $W_N$, $W_P$, $W_Q$, $W_R$ and $W_S$ from the matrices $Z$ and $W_Z$.\\It is called when a new value for $Z$ or $W_Z$ is given.\desc{See also}\funcName[@FWR/computeZ]{computeZ}\end{command}
