\begin{command}[@FWS/tf]{tf}	\desc{Purpose}Convert a FWS object into a tf object (transfer function)	\desc{Syntax}\matlab{H = tf(S,Te)}	\desc{Parameters}		\begin{tabular}{l@{\ :\ }p{9cm}}\matlab{H} &  tf object          \\\matlab{S} &  FWS object         \\\matlab{Te} &  period (default=1)\\		\end{tabular}	\desc{Description}Give the transfer function of the realization. The current realization (\matlab{R}) is considered.\\It is defined with matrices $A_Z$, $B_Z$, $C_Z$ and $D_Z$ (see eq. \eqref{eq:defAZandBZ} and \eqref{eq:defCZandDZ}) by:\begin{equation}H:z\mapsto C_Z \pa{zI_n-A_Z}^{-1}B_Z + D_Z\end{equation}\desc{See also}\funcName[@FWS/ss]{ss}, \funcName[@FWR/tf]{tf}\end{command}
