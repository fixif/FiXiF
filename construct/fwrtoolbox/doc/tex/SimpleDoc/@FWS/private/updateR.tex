\begin{command}[@FWS/updateR]{updateR}	\desc{Purpose}Update the R value from the (new) paramsValue	\desc{Syntax}\matlab{[R, cost\_flag] = updateR(S)}	\desc{Parameters}		\begin{tabular}{l@{\ :\ }p{9cm}}\matlab{R} &  FWR object\\\matlab{S} &  FWS object\\		\end{tabular}	\desc{Description}\begin{center}\I{Internal function}\end{center}This function updates the \matlab{R} value from the (new)parameters' values (\matlab{paramsValue}). This is done via the$\mt{U}$, $\mt{Y}$ and $\mt{W}$ matrices or directly, depending onthe FWS object and the \matlab{UYWfun} and \matlab{Runf}functions.\end{command}
