\begin{command}[strideM]{strideM}	\desc{Purpose}Create a stride permutation matrix (for FFT2FWR)	\desc{Syntax}\matlab{L = strideM(r,n)}	\desc{Parameters}		\begin{tabular}{l@{\ :\ }p{9cm}}\matlab{L} &  stride permutation matrix\\\matlab{r,n} &  parameters             \\		\end{tabular}	\desc{Description}Create a stride permutation matrix. This function is used by \funcName{FFT2FWR}.\\This matrix $\mathscr{L}^n_r$ is the stride permutation matrix that maps the vector elements indices $j$ as:\begin{equation}\mathscr{L}^{rs}_{r} : j \mapsto \begin{cases}jr \mod rs-1 &\text{for\ } 0 \leq j \leq rs - 2 \\rs-1 & \text{for\ } j=rs-1\end{cases}\end{equation}\desc{See also}\funcName[FFT2FWR]{FFT2FWR}\end{command}
