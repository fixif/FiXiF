\begin{command}[complexFFT]{complexFFT}	\desc{Purpose}Create FFT with complex coefficients (for FFT2FWR)(matrices J,L,N and S are used)	\desc{Syntax}\matlab{[J,L,N,S] = complexFFT(n, toSimplify)}	\desc{Parameters}		\begin{tabular}{l@{\ :\ }p{9cm}}\matlab{J,L,N,S} &  FWR matrices with complex coefficients                                       \\\matlab{n} &  size of the FFT                                                                    \\\matlab{toSimplify} &  1 (default) if the simplification (level=1) is done at each recursive step\\		\end{tabular}	\desc{Description}This function is called by \funcName{FFT2FWR}\desc{See also}\funcName[FFT2FWR]{FFT2FWR}, \funcName[@FWR/simplify]{simplify}, \funcName[strideM]{strideM}, \funcName[twiddleM]{twiddleM}\end{command}
